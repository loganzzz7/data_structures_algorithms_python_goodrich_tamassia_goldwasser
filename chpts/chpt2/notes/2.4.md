# 2.4 Inheritance
- A natural way to organize various components of a software package is in a `hierarchical` fashion
    - i.e. 
    ```
    building
        -> apartment 
            -> (low-rise, high-rise)
        -> house 
            -> (two-story house, ranch)
        -> commercial building 
            -> (skyscraper)
    ```
    - (apartment, house, commercial building) are a subset of building

- Inheritance -> there is a parent class then a child class
    - a subclass may `specialize` an existing behaviour by providing a new implementation that `overrides` an existing method or `extend` the parent class by providing new methods

class CreditCard:
    def __init__(self, customer, bank, acnt, limit):
        """
        create a new credit card instance

        init balance = zero

        customer -> name of customer
        bank -> name of the bank
        acnt -> the account identifier
        limit -> credit limit
        """
        self._customer = customer
        self._bank = bank
        self._account = acnt
        self._limit = limit
        self._balance = 0

    def get_customer(self):
        """return name of the customer"""
        return self._customer
    
    def get_bank(self):
        """return the bank's name"""
        return self._bank
    
    def get_account(self):
        """return the card identifying number (typically stored as a string)"""
        return self._account
    
    def get_limit(self):
        """return the current credit limit"""
        return self._limit
    
    def get_balance(self):
        """return the current balance"""
        return self._balance

    def charge(self, price):
        """
        charge given price to card
        
        return True if charge was processed
        False if declined
        """

        if price + self._balance > self._limit: #charge exceeds so false
            return False
        else:
            # else went thru so make charge & true
            self._balance += price
            return True
        
    def make_payment(self, amount):
        """
        process customer payment that reduces balance
        """
        self._balance -= amount

class PredatoryCreditCard(CreditCard): # CreditCard is parent class
    """
    an extension to CreditCard that compounds interest and fees
    """
    def __init__(self, customer, bank, acnt, limit, apr):
        """
        Create a new predatory credit card instance

        the initial balance is zero

        customer -> name of customer
        bank -> name of the bank
        acnt -> the account identifier
        limit -> credit limit
        apr -> annual percentage rate (i.e. 0.0825 for 8.25% APR)
        """
        super().__init__(customer, bank, acnt, limit) # this calls the constructor from the super class / parent class
        self._apr = apr
    
    def charge(self, price):
        """
        charge the given price to the card assuming sufficient credit limit
        
        ret True if charge processed else False
        """
        success = super().charge(price)
        if not success:
            self._balance += 5 #penalty if not enough money
        else:
            return success
    
    def process_month(self):
        """
        assess monthly interest on outstanding balance
        """
        if self._balance > 0:
            # if positive balance then add on apr
            monthly_factor = pow(1 + self._apr, 1/12)
            self._balance *= monthly_factor
